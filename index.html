Example: Angular
In app.js: 
            
$httpProvider.interceptors.push(function ($injector) {
    return {
        'request':     function (config) {
            var token = $injector('auth').getToken();
            if (token) {
                config.headers.Authorization = 'Bearer '
                    + token;
            }
            return config;
        }
    }
});
